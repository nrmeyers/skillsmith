import hmac
import hashlib
import os
from fastapi import FastAPI, Request, HTTPException
import stripe

app = FastAPI()

@app.post("/webhooks/stripe")
async def handle_stripe_webhook(request: Request):
    # Read the raw body bytes before any parsing
    body = await request.body()
    
    # Get the signature header
    sig_header = request.headers.get("Stripe-Signature")
    if not sig_header:
        raise HTTPException(status_code=401, detail="Missing Stripe-Signature header")
    
    try:
        # Use Stripe's helper to verify the signature and check timestamp tolerance
        stripe.Webhook.construct_event(
            payload=body,
            sig_header=sig_header,
            secret=os.environ["STRIPE_WEBHOOK_SECRET"],
            tolerance=300  # Default replay tolerance of 5 minutes
        )
    except stripe.error.SignatureVerificationError:
        raise HTTPException(status_code=400, detail="Invalid signature")
    
    return {"received": True}
